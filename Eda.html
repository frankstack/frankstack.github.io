In [12]:
ls_stopwords = stopwords.words('english')
lm = WordNetLemmatizer()

def text_prep(text:str):
    tokens = []
    for w in word_tokenize(text):
        w = w.lower()
        if ((w.isalpha()) and (w not in ls_stopwords)): tokens.append(lm.lemmatize(w))
    return ' '.join(tokens)
